### P1：编排层补充（orchestrator.js + 总纲 JSON）

#### 3.1.1 更新 `orchestrator.js`

在现有门禁引擎基础上新增三个能力：

**Handoff Schema（交接规范）**

```javascript
// 每个 Phase 的 handoff 定义
const HANDOFF_SCHEMA = {
  phase_1_to_2: { input: 'spec.json', output: 'path.json', validator: 'validatePathEndpoint' },
  phase_2_to_3: { input: 'path.json', output: 'nodes.json', validator: 'validateNodeRelevance' },
  phase_3_to_4: { input: 'nodes.json', output: 'interfaces.json', validator: 'validateSchemaMatch' },
  phase_4_to_5: { input: 'interfaces.json', output: 'leaf_nodes.json', validator: 'validateLeafNodes' },
  phase_5_to_6: { input: 'leaf_nodes.json', output: '最终产出', validator: 'validateBlueprint' },
  phase_6_to_7: { input: '最终产出', output: 'verification.json', validator: 'validateContent' },
  phase_7_to_8: { input: 'verification.json', output: 'health_check.json', validator: 'validateHealth' },
};

function validateHandoff(fromPhase, toPhase, taskDir) {
  const schema = HANDOFF_SCHEMA[`phase_${fromPhase}_to_${toPhase}`];
  if (!schema) return { valid: false, error: '未知的交接' };
  const inputExists = fs.existsSync(path.join(taskDir, schema.input));
  return { valid: inputExists, schema, inputExists };
}
```

**Timeout 配置**

```javascript
const PHASE_TIMEOUTS = {
  phase_1: 300000,  // 5 分钟（意图锚定）
  phase_2: 120000,  // 2 分钟（路径锚定）
  phase_3: 180000,  // 3 分钟（节点识别）
  phase_4: 180000,  // 3 分钟（接口契约）
  phase_5: 600000,  // 10 分钟（递归拆解）
  phase_6: 1800000, // 30 分钟（实现集成）
  phase_7: 300000,  // 5 分钟（通路验证）
  phase_8: 120000,  // 2 分钟（交付收尾）
};

function checkTimeout(phase, startTime) {
  const elapsed = Date.now() - startTime;
  const limit = PHASE_TIMEOUTS[phase] || 300000;
  if (elapsed > limit) {
    return { timedOut: true, elapsed, limit };
  }
  return { timedOut: false, elapsed, limit };
}
```

**Idempotency Key**

```javascript
function generateIdempotencyKey(taskId, phase) {
  const crypto = require('crypto');
  return crypto.createHash('sha256')
    .update(`${taskId}:${phase}:${Date.now()}`)
    .digest('hex').substring(0, 16);
}

function checkIdempotency(taskDir, phase, key) {
  const idemFile = path.join(taskDir, '.idempotency');
  if (!fs.existsSync(idemFile)) return { isDuplicate: false };
  const records = JSON.parse(fs.readFileSync(idemFile, 'utf-8'));
  const existing = records.find(r => r.phase === phase && r.key === key);
  return { isDuplicate: !!existing, existing };
}
```
